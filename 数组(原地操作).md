
# 283 移动零
    class Solution(object):
    def moveZeroes(self, nums):
        """
        :type nums: List[int]
        :rtype: None Do not return anything, modify nums in-place instead.
        """

        zero_nums = 0
        j = 0
        for i in range(len(nums)):
            if nums[i] == 0:
                zero_nums += 1
            else:
                nums[j] = nums[i]
                j += 1
        for i in range(j, j+zero_nums):
            nums[i] = 0
        return nums

        zero_nums 记0的个数
        j 真正移动的角标
        i 遍历数组
        双指针
