# 234 回文链表
	class Solution(object):
        def isPalindrome(self, head):
            """
            :type head: ListNode
            :rtype: bool
            """
            res_lst = []
            cur = head
            while cur:
                res_lst.append(cur.val)
                cur = cur.next
            r2 = res_lst[::-1]
            if res_lst == r2:
                return True
            else:
                return False
	通过遍历链表将值放入一个list，通过反转list进行判断
